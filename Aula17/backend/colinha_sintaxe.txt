app.post('/users', async (req, res) => {
  try {
    const conn = await connection;
    const { name, email } = req.body;

    const [result] = await conn.query(
      'INSERT INTO users (name, email) VALUES (?, ?)',
      [name, email]
    );

    res.status(201).json({
      id: result.insertId,
      name,
      email
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
